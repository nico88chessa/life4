SET(ITEMS
    "LineEditValidated"
    "UnmovableDialog"
    "UnmovableMainWindow"
    )

FOREACH(ITEM ${ITEMS})
    IF (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.ui")
        SET(UI_FILES ${UI_FILES} "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.ui")
    ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.ui")

    IF (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")
        SET(HEADERS ${HEADERS} "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")
    ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")

    IF (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")
        SET(SOURCES ${SOURCES} "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")
    ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")

    IF (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.qrc")
        SET(QRC_FILES ${QRC_FILES} "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.qrc")
    ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.qrc")

ENDFOREACH(ITEM)

SET(PROJECT_SOURCES ${PROJECT_SOURCES} ${SOURCES} CACHE INTERNAL "")
SET(PROJECT_HEADERS ${PROJECT_HEADERS} ${HEADERS} CACHE INTERNAL "")
SET(PROJECT_UI_FILES ${PROJECT_UI_FILES} ${UI_FILES} CACHE INTERNAL "")
SET(PROJECT_QRC_FILES ${PROJECT_QRC_FILES} ${QRC_FILES} CACHE INTERNAL "")
SET(PROJECT_QSS_FILES ${PROJECT_QSS_FILES} ${QSS_FILES} CACHE INTERNAL "")

MESSAGE("${CMAKE_CURRENT_SOURCE_DIR}/PROJECT_SOURCES: ${PROJECT_SOURCES}")
MESSAGE("${CMAKE_CURRENT_SOURCE_DIR}/PROJECT_HEADERS: ${PROJECT_HEADERS}")
MESSAGE("${CMAKE_CURRENT_SOURCE_DIR}/PROJECT_UI_FILES: ${PROJECT_UI_FILES}")
MESSAGE("${CMAKE_CURRENT_SOURCE_DIR}/QRC_FILES: ${PROJECT_QRC_FILES}")
MESSAGE("${CMAKE_CURRENT_SOURCE_DIR}/QSS_FILES: ${PROJECT_QSS_FILES}")
