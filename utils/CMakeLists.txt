SET(CLASS_NAMES
    "MathUtils"
    )

FOREACH(ITEM ${CLASS_NAMES})
    IF (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")
        SET(PROFILE_HEADERS ${PROFILE_HEADERS} "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")
    ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")

    IF (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")
        SET(PROFILE_SOURCES ${PROFILE_SOURCES} "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")
    ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")
ENDFOREACH(ITEM)

MESSAGE("PROFILE_HEADERS: ${PROFILE_HEADERS}")
MESSAGE("PROFILE_SOURCES: ${PROFILE_SOURCES}")

SET(PROJECT_SOURCES ${PROJECT_SOURCES} ${PROFILE_SOURCES} CACHE INTERNAL "")
SET(PROJECT_HEADERS ${PROJECT_HEADERS} ${PROFILE_HEADERS} CACHE INTERNAL "")
