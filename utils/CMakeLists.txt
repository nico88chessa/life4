SET(CLASS_NAMES
    "MathUtils"
    )

FOREACH(ITEM ${CLASS_NAMES})
    IF (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")
        SET(UTILS_HEADERS ${UTILS_HEADERS} "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")
    ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.hpp")

    IF (EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")
        SET(UTILS_SOURCES ${UTILS_SOURCES} "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")
    ENDIF(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/${ITEM}.cpp")
ENDFOREACH(ITEM)

MESSAGE("UTILS_HEADERS: ${UTILS_HEADERS}")
MESSAGE("UTILS_SOURCES: ${UTILS_SOURCES}")

SET(PROJECT_SOURCES ${PROJECT_SOURCES} ${UTILS_SOURCES} CACHE INTERNAL "")
SET(PROJECT_HEADERS ${PROJECT_HEADERS} ${UTILS_HEADERS} CACHE INTERNAL "")
